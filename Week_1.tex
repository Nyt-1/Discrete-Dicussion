\item[1)] Is the following satisfiable?
\begin{align*}
    (p \lor q) \land (\neg p \lor r) \land (\neg q \lor r) \land (\neg r \lor s) \land (\neg s)
\end{align*}

\item[2)] Simplify the following to only use $\neg , \lor, \land$
\begin{itemize}
    \item[i)] $(p \implies q) \iff (\neg q \implies \neg p)$
    \item[ii)] $\neg ((p \land \neg q) \lor (\neg r \land (s \lor \neg t)))$
    \item[iii)] $\neg((p \implies q) \land (\neg r \implies (s \lor t)))$
\end{itemize}


\item[3)] Explain the difference between $\forall x \exists y, f(x,y)$ and  $\forall y \exists x, f(x,y)$. Give an example where they are not equivalent.

\item[4)] Negate the following statement: 
\[\forall x \exists y,f(x,y) \land (x = y)\]

\item[5)] Prove $|A \times B| = |A|\cdot|B|$

\item[6)] True or False. If false, give a concrete example where it is false. If true, prove it.
\begin{itemize}
    \item[i)] $(A \times B) \cap (C \times D) = (A \cap C) \times (B \cap D)$
    \item[ii)] $(A \times B) \cup (C \times D) = (A \cup C) \times (B \cup D)$
    \item[iii)] $(A \setminus B) \cup C = (A \cup C) \setminus (B \cup C)$
\end{itemize}

\end{itemize}
